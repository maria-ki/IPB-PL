import string

alfa = string.ascii_lowercase
alfaLen = len(alfa)

def shift(text, shiftValue):
  newText = ""
  for i in range(len(text)):
    if ord(text[i]) + shiftValue <= 122:
      newText += chr(ord(text[i]) + shiftValue)
    else:
      newText += chr(ord(text[i]) + shiftValue - alfaLen)
  return newText
